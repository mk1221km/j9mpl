// Pure-memory ring buffer — no database, no I/O
type RingBuffer struct {
    Buf   [1024]float64
    Head  int
    Count int
}

func (rb *RingBuffer) Push(value float64) {
    rb.Buf[rb.Head] = value
    rb.Head = (rb.Head + 1) % len(rb.Buf)
    if rb.Count < len(rb.Buf) {
        rb.Count++
    }
}
