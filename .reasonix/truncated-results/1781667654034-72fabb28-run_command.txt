func (s *TransactionRouter) routeTransaction(dbPath string, record TransactionRecord) (string, error) {
	// Input validation: dbPath
	if strings.Contains(dbPath, "..") || strings.Contains(dbPath, "/etc/") {
		return "", fmt.Errorf("path traversal blocked")
	}
	// Input validation: record fields
	if strings.TrimSpace(record.TxId) == "" {
		return "", fmt.Errorf("invalid input: TxId is empty")
	}
	if strings.TrimSpace(record.Sender) == "" {
		return "", fmt.Errorf("invalid input: Sender is empty")
	}
	if strings.TrimSpace(record.Receiver) == "" {
		return "", fmt.Errorf("invalid input: Receiver is empty")
	}
	if strings.TrimSpace(record.Priority) == "" {
		return "", fmt.Errorf("invalid input: Priority is empty")
	}
	// SQL injection pattern checks
	checkFields := []string{record.TxId, record.Sender, record.Receiver, record.Priority}
	for _, f := range checkFields {
		if strings.ContainsAny(f, "'\"") || strings.Contains(f, ";") || strings.Contains(f, "--") || strings.Contains(f, "/*") {
			return "", fmt.Errorf("sql injection blocked")
		}
	}

	db, err := sql.Open("sqlite3", dbPath)
	if err != nil {
		return "", fmt.Errorf("open database: %w", err)
	}
	defer db.Close()

	// Determine channel from routing_rules
	var channel string
	err = db.QueryRow(
		"SELECT channel FROM routing_rules WHERE priority = ? AND min_amount <= ? ORDER BY min_amount DESC LIMIT 1",
		record.Priority, record.Amount,
	).Scan(&channel)
	if err != nil {
		if err == sql.ErrNoRows {
			return "", fmt.Errorf("no routing rule found for priority %s and amount %.2f", record.Priority, record.Amount)
		}
		return "", fmt.Errorf("query routing_rules: %w", err)
	}

	// Insert into transaction_log
	_, err = db.Exec(
		"INSERT INTO transaction_log (tx_id, sender, receiver, amount, channel, status) VALUES (?, ?, ?, ?, ?, 'ROUTED')",
		record.TxId, record.Sender, record.Receiver, record.Amount, channel,
	)
	if err != nil {
		return "", fmt.Errorf("insert transaction_log: %w", err)
	}

	return channel, nil
}

func (s *TransactionRouter) getTransactionCount(dbPath string, status string) (int, error) {
	// --- input validation ---
	if strings.Contains(dbPath, "..") || strings.Contains(dbPath, "/etc/") {
		return 0, fmt.Errorf("path traversal blocked")
	}
	if strings.TrimSpace(status) == "" {
		return 0, fmt.Errorf("invalid input: status is empty")
	}
	// simple SQL injection keyword/character check
	if strings.ContainsAny(status, "'\"\\;-") || strings.Contains(status, "--") {
		return 0, fmt.Errorf("sql injection blocked")

	// --- database operation ---
	db, err := sql.Open("sqlite3", dbPath)

}

func main() error {
    db, err := sql.Open("sqlite3", "./transactions.db")

        return fmt.Errorf("failed to open database: %w", err)
    }
    defer db.Close()

    if _, err := db.Exec(`

        );
        CREATE TABLE IF NOT EXISTS transaction_log (
            tx_id TEXT PRIMARY KEY,

            receiver TEXT,
            amount REAL,
            channel TEXT,
            status TEXT
        );
    `); err != nil {
        return fmt.Errorf("failed to create tables: %w", err)
    }

    if err := router.InitRoutingTable("default routing rules configuration"); err != nil {
        return fmt.Errorf("failed to init routing table: %w", err)
    }

        {TxId: "TXN001", Sender: "Alice", Receiver: "Bob", Amount: 100.0, Priority: "high"},

        {TxId: "TXN003", Sender: "Eve", Receiver: "Frank", Amount: 200.0, Priority: "low"},
    }

    for _, txn := range testTransactions {
        if err := router.RouteTransaction("default", txn); err != nil {
            return fmt.Errorf("failed to route transaction %s: %w", txn.TxId, err)
        }
    }

    totalCount, err := router.GetTransactionCount("", "")
    if err != nil {
        return fmt.Errorf("failed to get total transaction count: %w", err)
    }
    expectedTotal := len(testTransactions)
    if totalCount != expectedTotal {
        return fmt.Errorf("expected %d total transactions, got %d", expectedTotal, totalCount)
    }

    defaultCount, err := router.GetTransactionCount("default", "")
    if err != nil {
        return fmt.Errorf("failed to get default channel count: %w", err)
    }
    if defaultCount != expectedTotal {
        return fmt.Errorf("expected %d transactions in default channel, got %d", expectedTotal, defaultCount)
    }

    return nil
}
